# Data Schemas

## Available Schemas

### AllowedUserRoles

```json
{
  "description": "Roles permitidos en el sistema GeoterRA",
  "type": "string",
  "enum": [
    "admin",
    "maintenance",
    "user"
  ]
}
```

### AnalysisRequestDTO

```json
{
  "description": "Solicitud de análisis de punto geotérmico",
  "required": [
    "region",
    "email",
    "temperature_sensation",
    "latitude",
    "longitude"
  ],
  "properties": {
    "region": {
      "description": "ID de la región",
      "type": "integer",
      "example": 1
    },
    "email": {
      "description": "Email de contacto del solicitante",
      "type": "string",
      "format": "email",
      "example": "contact@example.com"
    },
    "owner_contact_number": {
      "description": "Número de contacto del dueño",
      "type": "string",
      "example": "87654321",
      "nullable": true
    },
    "owner_name": {
      "description": "Nombre del dueño del terreno",
      "type": "string",
      "example": "Carlos Mendoza",
      "nullable": true
    },
    "temperature_sensation": {
      "description": "Percepción térmica del lugar",
      "type": "string",
      "example": "Cálido",
      "enum": [
        "Muy frío",
        "Frío",
        "Templado",
        "Cálido",
        "Muy Caliente",
        "caliente"
      ]
    },
    "bubbles": {
      "description": "¿Hay burbujas de gas?",
      "type": "boolean",
      "example": false
    },
    "details": {
      "description": "Detalles adicionales del punto",
      "type": "string",
      "example": "Zona con actividad geotérmica notable",
      "nullable": true
    },
    "current_usage": {
      "description": "Uso actual del terreno",
      "type": "string",
      "example": "Agrícola",
      "nullable": true
    },
    "latitude": {
      "description": "Latitud del punto en grados decimales",
      "type": "number",
      "format": "float",
      "example": 10.463
    },
    "longitude": {
      "description": "Longitud del punto en grados decimales",
      "type": "number",
      "format": "float",
      "example": -85.4519
    },
    "state": {
      "description": "Estado actual de la solicitud",
      "type": "string",
      "example": "Registrada",
      "nullable": true,
      "enum": [
        "Registrada",
        "En revisión",
        "Verificación de campo",
        "Análisis en laboratorio",
        "Aprobada",
        "Rechazada",
        "Archivada"
      ]
    }
  },
  "type": "object"
}
```
